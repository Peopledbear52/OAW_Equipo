Código para obtener toda la información de una noticia teniendo la url de su feed

SELECT 
    n.titulo, 
    n.fecha, 
    n.descripcion, 
    f.titulo AS feed_nombre, 
    c.nombre AS categoria
FROM 
    noticias n
JOIN 
    feeds f ON n.url = f.url
LEFT JOIN 
    noticias_categorias nc ON n.id = nc.noticia_id
LEFT JOIN 
    categorias c ON nc.categoria_id = c.id
WHERE 
    n.url = 'http://ejemplo.com/feed';
